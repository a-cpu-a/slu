#
# Dependencies: name, numbers
#

New Syntax {
	
	globstat ::= ";"
	--TODO: ban non global statements in chunks at parse time.
	chunk ::= {globstat}

	basicstat ::= ";"
		| var "=" expr
		| var selfablecall
	
	statOrExpr ::=
	
	--TODO: consider making pointers opaque: type only contains alignment

	block ::= {stat} [retstat [";"]]
	retstat ::= ("return" | ("break" | "continue") ["'" Name]) [expr]
	blockOrRet ::= retstat [";"] | "{" block "}"
	
	label ::= "'" Name ":"
	blockstat ::= [label] "{" block "}"
	stat ::= basicstat
		| globstat
		| statOrExpr
		| blockstat
	
	tupleableName ::= Name | integral
	subvar ::= ".*" | "." tupleableName | "[" expr "]"
	var ::= (Name | "(" expr ")") {{selfablecall} subvar}

	basicArgs ::=  "(" [expr {"," expr} [","]] ")" | LiteralString | Numeral
	args ::=  basicArgs | tableconstructor
	selfablecallBasic ::= ["." Name] basicArgs
	selfablecall ::= ["." Name] args

	basicExpr ::=  Name | "(" expr ")" | LiteralString | Numeral | tableconstructor
		| basicExpr subvar | basicExpr selfablecallBasic
		| basicExpr binop basicExpr | {preop} basicExpr {sufop}
		| statOrExpr
	
	expr ::=  basicExpr
		| expr subvar | expr selfablecall
		| expr binop expr | {preop} expr {sufop}

	fieldsep ::= ["," | ";"]
	field ::= (Name "=" expr) | expr
	tableconstructor ::= "{" {field fieldsep} "}"

	binop ::=  "+" | "-" | "*" | "/" | "//" | "^" | "%"
		| "++"
		| "<" | "<=" | ">" | ">=" | "==" | "!="
		| "and" | "or"

	preop ::= "-" | "!"
	sufop ::=
}

New Keywords {
	and
	break
	continue
	or
	return
}

Basic {
	If the user forgets to add a `;` or `,`, then its a warning.
	`;` for statements, match and try.
	`,` for tables.
}
