}

New Abis {
	
	"C" {
		Compatible with c code
	}
	"C++" {
		Compatible with c++ code
	}
	"Slu" {
		By default Slu is equivalent to Any.
		However, if dynamic linking is used, and this is exported out of the crate, then will be equivalent to SluLock.
	}
	"SluLock" {
		A abi will be generated & optimized for the thing(func,var,...) & put into some lock file.
	}
	"Lua" {
		Compatible with lua code
	}
	"Js" {
		Compatible with js code
	}
	"Java" {
		Compatible with java code
	}
	"Any" {
		No defined abi, useful in dynamicly linked libraries, where you need a exported thing, with no defined abi.
		The abi may change at any moment. (even while the program is running... maybe (smc))
		
		If a function pointer is taken, a wrapper function may be created sometimes, to allow all functions to have the same abi.
		
		Since the abi is undefined, this could just be equivalent to "C" with some arg manipulation.
	}
	
}

Extern {
	
	Imports functions from the host environment.
	
	Can be used to call things like C functions, or lua functions.
	
	C pointers -> usize, or cc::IntPtr(T)
	C const pointers -> usize, or cc::IntPtr(T)
	
	C arrays -> {T}**N (? can they be passed by value like this ?)
	
	cpp const refs -> Ptr(T)
	cpp refs -> PtrShare(T)
	
	cpp && refs -> likely just: PtrMut(T) or PtrShare(T)
	
	lua any-type		-> lua_ext::Any
	lua integers		-> lua_ext::integer
	lua numbers 		-> lua_ext::number
	lua strings 		-> lua_ext::LuaStr
	lua tables  		-> lua_ext::LuaTable(...)
	lua functions		-> lua_ext::LuaFn(...)
	lua threads  		-> lua_ext::LuaThread
	lua user-data		-> lua_ext::LuaUserData(...)
	lua light-user-data	-> lua_ext::LuaLightData
	lua nil				-> lua_ext::LuaNil
	
	js any-type 	-> js_ext::Any
	js integers 	-> js_ext::integer
	js numbers		-> js_ext::number
	js string		-> js_ext::JsStr
	js object		-> js_ext::JsObject(...)
	js array		-> js_ext::JsArray(...)
	js functions	-> js_ext::JsFn(...)
	js symbol		-> js_ext::JsSymbol
	js null			-> js_ext::JsNull
	js undefined	-> js_ext::JsUndefined

	}
}
