New Syntax {
	
	safety ::= "safe" | "unsafe"
	
	stat @::= | "unsafe" "{" {stat} "}"
	stat @::= | ":::" safety ":"
	
}

New Keywords {
	safe
	unsafe
}

Unsafe Block {
	Inside a unsafe block, there is no new scope, so variables are usable even after the block ends, due to this return/break is not allowed inside unsafe blocks, also inside it any unsafe code can be used.
}
Safety Label {
	After the label, the current safety mode switches.
	Cant be used inside of a unsafe block.
}
