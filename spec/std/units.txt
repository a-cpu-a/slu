Unit Type {
	Stores a list of units, and their integer exponents
}

Quant Type {
	number(u8 | f32 | 1 | 100 | 0.1 | 1/3 | ... etc) + unit
}

Units {
	std::units
	```slu
	struct BitTag{}; 
	ex const Bit = Unit.:new(Data, BitTag);
	ex const bit = Quant.:new(|| 1, Bit);
	
	ex const byte = 8 * bit;
	```

	std::si
	```slu
	struct MeterTag{}; 
	ex const Meter = Unit.:new(Length, MeterTag);
	ex const m = Quant.:new(|| 1, Meter);

	ex const quetta = ronna * kilo;
	ex const ronna = yotta * kilo;
	ex const yotta = zetta * kilo;
	ex const zetta = exa * kilo;
	ex const exa = peta * kilo;
	ex const peta = tera * kilo;
	ex const tera = giga * kilo;
	ex const giga = mega * kilo;
	ex const mega = kilo * kilo;
	ex const kilo = hecto * deca;
	ex const hecto = deca * deca;
	ex const deca = 10;
	
	ex const deci = 1 / deca;
	ex const centi = 1 / hecto;
	ex const milli = 1 / kilo;
	ex const micro = 1 / mega;
	ex const nano = 1 / giga;
	ex const pico = 1 / tera;
	ex const femto = 1 / peta;
	ex const atto = 1 / exa;
	ex const zepto = 1 / zetta;
	ex const yocto = 1 / yotta;
	ex const ronto = 1 / ronna;
	ex const quecto = 1 / quetta;
	```
}

Archive {
	-- 1. The unique tag (phantom type) to identify "Meter"
	struct MeterTag{}; 

	-- This creates the Unit definition. 
	-- Type: Unit. Data: {Units: [{Base: Length, Tag: MeterTag, Coeff: 1}]}
	ex const Meter = Unit.:new(Length, MeterTag);

	-- This creates a specific value using that Unit.
	-- Type: Quant. Data: {Value: 1, Unit: Meter}
	ex const m = Quant.:new(1, Meter);

	-- Complex derived unit example
	-- Type: Quant. Data: {Value: 1, Unit: (Meter / Second)}
	ex const mps = Quant.:new(1, Meter / Second);
}
